#!/bin/bash
# Text editors - Ejercicio 1

# usando ps, se listan todos los procesos del sistema
ps -x

# usando ps y grep, se listan todos los procesos del usuario actual
ps -xu | grep -i "e516985"

# se listan los 10 procesos más recientes del usuario actual
ps -xu | grep -i "e516985" | tail -n 10

# se muestran las páginas de manual de grep usando como máximo 100 caracteres por línea
man grep | fmt -100 -s

# se realiza el mismo proceso anterior, pero sustituyendo cualquier dígito por un asterisco (*)
man grep | fmt -100 -s | tr '0-9' '*'

# se realiza lo mismo que antes, pero enviando el resultado a un fichero
man grep | fmt -100 -s | tr '0-9' '*' > man_grep_asterisk.txt
